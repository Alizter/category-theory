Class Product (C : Category) (P : C) `(p1 : P ~> A) `(p2 : P ~> B) :=
{ product_ump :
    ∀ (X : C) (x1 : X ~> A) (x2 : X ~> B),
       ∃ (u : X ~> P), x1 = p1 ∘ u ∧ x2 = p2 ∘ u
    ∧ ∀ (v : X ~> P), p1 ∘ v = x1 ∧ p2 ∘ v = x2 → v = u
}.

Set Printing All.
Set Printing Universes.
Program Instance Pair {X Y : Set}
  : Product Sets (X × Y) (@fst X Y) (@snd X Y).
Obligation 1.   ∃ (fun x ⇒ (x1 x, x2 x)).
  intros. constructor.
    intros. unfold fst. reflexivity.
  split.
    intros. unfold snd. reflexivity.
  intros.
  inversion H.
  extensionality e.
  rewrite <- H0.
  rewrite <- H1.
  destruct (v e).
  reflexivity.
Defined.

Definition Tuple_map {Z X Y} (f : X → Y) (p : Z × X) : Z × Y :=
  match p with
  | pair z x ⇒ @pair Z Y z (f x)
  end.

Program Instance Tuple_Functor {Z} : Arr ⟶ Arr :=
{ fobj := fun X ⇒ Z × X
; fmap := @Tuple_map Z
}.
